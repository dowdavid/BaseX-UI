import { RadioBasic, RadioHorizontal, RadioDisabled } from '@/components/demos/radio-demo';

## Import

```tsx
import { Radio } from '@basex-ui/components';
```

## Usage

<RadioBasic />

```tsx
<Radio.Group defaultValue="apple">
  <label><Radio.Root value="apple"><Radio.Indicator /></Radio.Root> Apple</label>
  <label><Radio.Root value="banana"><Radio.Indicator /></Radio.Root> Banana</label>
  <label><Radio.Root value="cherry"><Radio.Indicator /></Radio.Root> Cherry</label>
</Radio.Group>
```

## Horizontal

<RadioHorizontal />

```tsx
<Radio.Group defaultValue="left" orientation="horizontal">
  <label><Radio.Root value="left"><Radio.Indicator /></Radio.Root> Left</label>
  <label><Radio.Root value="center"><Radio.Indicator /></Radio.Root> Center</label>
  <label><Radio.Root value="right"><Radio.Indicator /></Radio.Root> Right</label>
</Radio.Group>
```

## Parts

| Part | Description |
|------|-------------|
| `Group` | Container managing selection |
| `Root` | Individual radio button (requires `value`) |
| `Indicator` | Selected dot |

## Props

### Group

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | — | Controlled selection |
| `defaultValue` | `string` | — | Initially selected |
| `onValueChange` | `(value) => void` | — | Called on change |
| `orientation` | `'vertical' \| 'horizontal'` | `'vertical'` | Layout direction |
| `disabled` | `boolean` | `false` | Disable all radios |

### Root

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | **required** | Value for this option |
| `disabled` | `boolean` | `false` | Disable this radio |
