---
title: Menubar
description: A horizontal container for multiple menus with keyboard navigation.
---

import { MenubarBasic } from '@/components/demos/menubar-demo';

## Import

```tsx
import { Menu, Menubar } from '@basex-ui/components';
```

## Usage

<MenubarBasic />

```tsx
<Menubar>
  <Menu.Root>
    <Menu.Trigger>File</Menu.Trigger>
    <Menu.Portal>
      <Menu.Positioner>
        <Menu.Popup>
          <Menu.Item>New</Menu.Item>
          <Menu.Item>Open</Menu.Item>
          <Menu.Item>Save</Menu.Item>
        </Menu.Popup>
      </Menu.Positioner>
    </Menu.Portal>
  </Menu.Root>
</Menubar>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `modal` | `boolean` | `true` | Block interaction behind open menus |
| `disabled` | `boolean` | `false` | Disable all menus |
| `orientation` | `'horizontal' \| 'vertical'` | `'horizontal'` | Layout direction |
| `loopFocus` | `boolean` | `true` | Loop keyboard focus |
