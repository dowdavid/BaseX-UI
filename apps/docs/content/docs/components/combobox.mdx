---
title: Combobox
description: A searchable select dropdown. Users pick from a predefined list with optional multi-select.
---

import { ComboboxBasic, ComboboxMulti } from '@/components/demos/combobox-demo';

## Import

```tsx
import { Combobox } from '@basex-ui/components';
```

## Usage

<ComboboxBasic />

```tsx
const fruits = ['Apple', 'Banana', 'Cherry', 'Date', 'Elderberry'];

<Combobox.Root items={fruits}>
  <Combobox.Input placeholder="Select a fruit..." />
  <Combobox.Popup>
    {(item) => (
      <Combobox.Item value={item}>
        <Combobox.ItemIndicator />
        {item}
      </Combobox.Item>
    )}
  </Combobox.Popup>
</Combobox.Root>
```

## Multi-select

<ComboboxMulti />

```tsx
<Combobox.Root items={fruits} multiple>
  <Combobox.Input placeholder="Select fruits..." />
  <Combobox.Popup>
    {(item) => (
      <Combobox.Item value={item}>
        <Combobox.ItemIndicator />
        {item}
      </Combobox.Item>
    )}
  </Combobox.Popup>
</Combobox.Root>
```

## Parts

| Part | Description |
|------|-------------|
| `Root` | Container managing items and selection (requires `items`) |
| `Input` | Searchable text input |
| `Popup` | Dropdown container |
| `Item` | Selectable option (requires `value`) |
| `ItemIndicator` | Checkmark for selected items |
| `Clear` | Clears the selection |
| `Empty` | Shown when no items match |

## Props

### Root

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `items` | `string[]` | **required** | List of options |
| `value` | `string \| string[]` | — | Controlled selection |
| `onValueChange` | `(value) => void` | — | Called on selection |
| `multiple` | `boolean` | `false` | Allow multiple selections |
| `size` | `'sm' \| 'md' \| 'lg'` | `'md'` | Input size |
| `disabled` | `boolean` | `false` | Disabled |
