---
title: Accordion
description: Collapsible panels with headings for progressive content disclosure.
---

import { AccordionBasic, AccordionMultiple } from '@/components/demos/accordion-demo';

## Import

```tsx
import { Accordion } from '@basex-ui/components';
```

## Usage

<AccordionBasic />

```tsx
<Accordion.Root defaultValue={['item-1']}>
  <Accordion.Item value="item-1">
    <Accordion.Header>
      <Accordion.Trigger>What is BaseX UI?</Accordion.Trigger>
    </Accordion.Header>
    <Accordion.Panel>
      BaseX UI is an AI-first, copy-paste component library built on Base UI and styled with StyleX.
    </Accordion.Panel>
  </Accordion.Item>
</Accordion.Root>
```

## Multiple panels

<AccordionMultiple />

```tsx
<Accordion.Root multiple defaultValue={['item-1', 'item-2']}>
  <Accordion.Item value="item-1">
    <Accordion.Header>
      <Accordion.Trigger>Section 1 (open)</Accordion.Trigger>
    </Accordion.Header>
    <Accordion.Panel>Both panels can be open at the same time.</Accordion.Panel>
  </Accordion.Item>
</Accordion.Root>
```

## Parts

| Part | Description |
|------|-------------|
| `Root` | Container managing open state |
| `Item` | Individual collapsible section (requires `value`) |
| `Header` | Wrapper for the trigger |
| `Trigger` | Button that toggles the panel |
| `Panel` | Collapsible content area |

## Props

### Root

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string \| string[]` | — | Controlled open items |
| `defaultValue` | `string \| string[]` | — | Initially open items |
| `onValueChange` | `(value) => void` | — | Called when open items change |
| `multiple` | `boolean` | `false` | Allow multiple panels open |
| `disabled` | `boolean` | `false` | Disable all items |

### Item

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | **required** | Unique identifier |
| `disabled` | `boolean` | `false` | Disable this item |
