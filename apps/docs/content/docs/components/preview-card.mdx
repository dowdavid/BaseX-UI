import { PreviewCardBasic } from '@/components/demos/preview-card-demo';

## Import

```tsx
import { PreviewCard } from '@basex-ui/components';
```

## Usage

<PreviewCardBasic />

```tsx
<PreviewCard.Root>
  <PreviewCard.Trigger href="https://basex-ui.dev">
    Base-X UI
  </PreviewCard.Trigger>
  <PreviewCard.Portal>
    <PreviewCard.Positioner>
      <PreviewCard.Popup>
        <h3>Base-X UI</h3>
        <p>An agent-first component library.</p>
      </PreviewCard.Popup>
    </PreviewCard.Positioner>
  </PreviewCard.Portal>
</PreviewCard.Root>
```

## Parts

| Part | Description |
|------|-------------|
| `Root` | Controls hover state |
| `Trigger` | Anchor element that triggers the card |
| `Portal` | Renders outside the DOM tree |
| `Positioner` | Positions relative to trigger |
| `Popup` | Card content |
| `Arrow` | Arrow pointing to trigger |

## Props

### Root

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `open` | `boolean` | — | Controlled open state |
| `defaultOpen` | `boolean` | `false` | Initially open |
| `onOpenChange` | `(open) => void` | — | Called on change |
| `delay` | `number` | `600` | Open delay in ms |
| `closeDelay` | `number` | `300` | Close delay in ms |
