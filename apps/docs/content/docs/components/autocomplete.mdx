import { AutocompleteBasic, AutocompleteGrouped, AutocompleteAutoHighlight, AutocompleteStartAddon, AutocompleteSizes } from '@/components/demos/autocomplete-demo';

## Import

```tsx
import { Autocomplete } from '@basex-ui/components';
```

## Usage

<AutocompleteBasic />

```tsx
const fruits = [
  { id: 1, value: 'Apple' },
  { id: 2, value: 'Banana' },
  { id: 3, value: 'Cherry' },
  // ...
];

<Autocomplete.Root items={fruits}>
  <Autocomplete.Input placeholder="Search fruits..." />
  <Autocomplete.Popup>
    <Autocomplete.Empty>No fruits found.</Autocomplete.Empty>
    {(fruit) => (
      <Autocomplete.Item key={fruit.id} value={fruit}>
        {fruit.value}
      </Autocomplete.Item>
    )}
  </Autocomplete.Popup>
</Autocomplete.Root>
```

## Grouped suggestions

<AutocompleteGrouped />

```tsx
<Autocomplete.Root items={produce}>
  <Autocomplete.Input placeholder="Search produce..." />
  <Autocomplete.Popup>
    <Autocomplete.Empty>No results.</Autocomplete.Empty>
    {(group) => (
      <Autocomplete.Group key={group.label}>
        <Autocomplete.GroupLabel>{group.label}</Autocomplete.GroupLabel>
        {group.items.map((item) => (
          <Autocomplete.Item key={item.id} value={item}>
            {item.value}
          </Autocomplete.Item>
        ))}
      </Autocomplete.Group>
    )}
  </Autocomplete.Popup>
</Autocomplete.Root>
```

## Parts

| Part | Description |
|------|-------------|
| `Root` | Container managing items and filtering (requires `items`) |
| `Input` | Text input with optional clear button |
| `Popup` | Dropdown container |
| `Item` | Individual suggestion (requires `value`) |
| `Empty` | Shown when no items match |
| `Group` | Groups items under a label |
| `GroupLabel` | Label for a group |

## Props

### Root

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `items` | `string[]` | **required** | List of suggestion items |
| `value` | `string` | — | Controlled selected value |
| `onValueChange` | `(value) => void` | — | Called on selection |
| `size` | `'sm' \| 'md' \| 'lg'` | `'md'` | Input size |
| `disabled` | `boolean` | `false` | Disable the input |
