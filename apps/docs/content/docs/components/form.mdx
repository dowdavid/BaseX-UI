title: Form

import { FormBasic } from '@/components/demos/form-demo';

## Import

```tsx
import { Form, Field, Input, Button } from '@basex-ui/components';
```

## Usage

<FormBasic />

```tsx
<Form.Root onSubmit={(e) => e.preventDefault()}>
  <Fieldset.Root>
    <Fieldset.Legend>Contact</Fieldset.Legend>
    <Field.Root>
      <Field.Label>Name</Field.Label>
      <Input />
    </Field.Root>
    <Field.Root>
      <Field.Label>Email</Field.Label>
      <Input type="email" />
    </Field.Root>
  </Fieldset.Root>
  <Button type="submit">Submit</Button>
</Form.Root>
```

## Props

### Root

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `errors` | `Record<string, string[] \| null>` | — | Server-side validation errors by field name |
| `validationMode` | `'onBlur' \| 'onChange' \| 'onSubmit'` | — | When to validate |
| `onFormSubmit` | `(data) => void` | — | Called on submit |
