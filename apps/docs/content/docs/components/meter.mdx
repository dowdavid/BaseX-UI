import { MeterBasic, MeterWithValue, MeterColors, MeterSizes } from '@/components/demos/meter-demo';

## Import

```tsx
import { Meter } from '@basex-ui/components';
```

## Usage

<MeterBasic />

```tsx
<Meter.Root value={65}>
  <Meter.Label>Storage used</Meter.Label>
  <Meter.Track>
    <Meter.Indicator />
  </Meter.Track>
</Meter.Root>
```

## With value display

<MeterWithValue />

```tsx
<Meter.Root value={42}>
  <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center' }}>
    <Meter.Label>Upload progress</Meter.Label>
    <Meter.Value />
  </div>
  <Meter.Track>
    <Meter.Indicator />
  </Meter.Track>
</Meter.Root>
```

## Parts

| Part | Description |
|------|-------------|
| `Root` | Container with value context (requires `value`) |
| `Label` | Accessible label |
| `Track` | Background bar |
| `Indicator` | Filled portion |
| `Value` | Formatted value text |

## Props

### Root

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `number` | **required** | Current value |
| `min` | `number` | `0` | Minimum |
| `max` | `number` | `100` | Maximum |

### Track

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `size` | `'sm' \| 'md' \| 'lg'` | `'md'` | Track height |

### Indicator

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `color` | `'default' \| 'secondary' \| 'destructive'` | `'default'` | Bar color |
