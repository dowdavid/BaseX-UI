import { NumberFieldBasic, NumberFieldConstrained, NumberFieldSizes, NumberFieldDisabled } from '@/components/demos/number-field-demo';

## Import

```tsx
import { NumberField } from '@basex-ui/components';
```

## Usage

<NumberFieldBasic />

```tsx
<NumberField.Root>
  <NumberField.Group>
    <NumberField.Decrement />
    <NumberField.Input />
    <NumberField.Increment />
  </NumberField.Group>
</NumberField.Root>
```

## Constrained

<NumberFieldConstrained />

```tsx
<NumberField.Root defaultValue={0} min={0} max={100} step={5}>
  <NumberField.Group>
    <NumberField.Decrement />
    <NumberField.Input />
    <NumberField.Increment />
  </NumberField.Group>
</NumberField.Root>
```

<NumberFieldDisabled />

<NumberField.Root defaultValue={3} disabled>

## Parts

| Part | Description |
|------|-------------|
| `Root` | Container with value state |
| `Group` | Visual wrapper for input and buttons |
| `Input` | Numeric input |
| `Increment` | Plus button |
| `Decrement` | Minus button |

## Props

### Root

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `number` | — | Controlled value |
| `defaultValue` | `number` | — | Initial value |
| `onValueChange` | `(value) => void` | — | Called on change |
| `min` | `number` | — | Minimum value |
| `max` | `number` | — | Maximum value |
| `step` | `number` | `1` | Step increment |
| `disabled` | `boolean` | `false` | Disabled |

### Group

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `size` | `'sm' \| 'md' \| 'lg'` | `'md'` | Field size |
