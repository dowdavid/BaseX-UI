import { MenuBasic, MenuGroups, MenuCheckbox } from '@/components/demos/menu-demo';

## Import

```tsx
import { Menu } from '@basex-ui/components';
```

## Usage

<MenuBasic />

```tsx
<Menu.Root>
  <Menu.Trigger>Open Menu</Menu.Trigger>
  <Menu.Portal>
    <Menu.Positioner>
      <Menu.Popup>
        <Menu.Item onClick={() => {}}>Edit</Menu.Item>
        <Menu.Item onClick={() => {}}>Duplicate</Menu.Item>
        <Menu.Separator />
        <Menu.Item destructive onClick={() => {}}>
          Delete
        </Menu.Item>
      </Menu.Popup>
    </Menu.Positioner>
  </Menu.Portal>
</Menu.Root>
```

## Checkbox Items

<MenuCheckbox />

```tsx
const [showGrid, setShowGrid] = useState(false);
const [showRulers, setShowRulers] = useState(true);

<Menu.Root>
  <Menu.Trigger>Open Menu</Menu.Trigger>
  <Menu.Portal>
    <Menu.Positioner>
      <Menu.Popup>
        <Menu.CheckboxItem checked={showGrid} onCheckedChange={setShowGrid}>
          <Menu.CheckboxItemIndicator>✓</Menu.CheckboxItemIndicator>
          Show Grid
        </Menu.CheckboxItem>
        <Menu.CheckboxItem checked={showRulers} onCheckedChange={setShowRulers}>
          <Menu.CheckboxItemIndicator>✓</Menu.CheckboxItemIndicator>
          Show Rulers
        </Menu.CheckboxItem>
      </Menu.Popup>
    </Menu.Positioner>
  </Menu.Portal>
</Menu.Root>
```

## Parts

| Part | Description |
|------|-------------|
| `Root` | Controls open state |
| `Trigger` | Opens the menu |
| `Portal` | Renders outside the DOM tree |
| `Positioner` | Positions the popup |
| `Popup` | Menu container |
| `Item` | Menu action |
| `Separator` | Divider line |
| `Group` | Groups items under a label |
| `GroupLabel` | Label for a group |
| `CheckboxItem` | Toggleable item |
| `RadioGroup` | Radio group container |
| `RadioItem` | Exclusive selection item |

## Props

### Item

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `destructive` | `boolean` | `false` | Red destructive style |
| `closeOnClick` | `boolean` | `true` | Close menu on click |
