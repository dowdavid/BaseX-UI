import { MenuBasic } from '@/components/demos/menu-demo';

## Import

```tsx
import { Menu } from '@basex-ui/components';
```

## Usage

<MenuBasic />

```tsx
<Menu.Root>
  <Menu.Trigger>Actions</Menu.Trigger>
  <Menu.Portal>
    <Menu.Positioner>
      <Menu.Popup>
        <Menu.Item>Edit</Menu.Item>
        <Menu.Item>Duplicate</Menu.Item>
        <Menu.Separator />
        <Menu.Item destructive>Delete</Menu.Item>
      </Menu.Popup>
    </Menu.Positioner>
  </Menu.Portal>
</Menu.Root>
```

## Parts

| Part | Description |
|------|-------------|
| `Root` | Controls open state |
| `Trigger` | Opens the menu |
| `Portal` | Renders outside the DOM tree |
| `Positioner` | Positions the popup |
| `Popup` | Menu container |
| `Item` | Menu action |
| `Separator` | Divider line |
| `Group` | Groups items under a label |
| `GroupLabel` | Label for a group |
| `CheckboxItem` | Toggleable item |
| `RadioGroup` | Radio group container |
| `RadioItem` | Exclusive selection item |

## Props

### Item

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `destructive` | `boolean` | `false` | Red destructive style |
| `closeOnClick` | `boolean` | `true` | Close menu on click |
