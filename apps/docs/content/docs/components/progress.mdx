import { ProgressBasic, ProgressWithValue, ProgressColors, ProgressSizes, ProgressIndeterminate } from '@/components/demos/progress-demo';

## Import

```tsx
import { Progress } from '@basex-ui/components';
```

## Usage

<ProgressBasic />

```tsx
<Progress.Root value={65}>
  <Progress.Label>Uploading...</Progress.Label>
  <Progress.Track>
    <Progress.Indicator />
  </Progress.Track>
</Progress.Root>
```

## With value display

<ProgressWithValue />

```tsx
<Progress.Root value={42}>
  <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center' }}>
    <Progress.Label>Processing</Progress.Label>
    <Progress.Value />
  </div>
  <Progress.Track>
    <Progress.Indicator />
  </Progress.Track>
</Progress.Root>
```

## Indeterminate

Pass `value={null}` for an indeterminate progress bar.

<ProgressIndeterminate />

```tsx
<Progress.Root value={null}>
  <Progress.Label>Loading...</Progress.Label>
  <Progress.Track>
    <Progress.Indicator />
  </Progress.Track>
</Progress.Root>
```

## Parts

| Part | Description |
|------|-------------|
| `Root` | Container with value context |
| `Label` | Accessible label |
| `Track` | Background bar |
| `Indicator` | Filled portion |
| `Value` | Formatted value text |

## Props

### Root

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `number \| null` | **required** | Current value (`null` for indeterminate) |
| `min` | `number` | `0` | Minimum |
| `max` | `number` | `100` | Maximum |

### Track

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `size` | `'sm' \| 'md' \| 'lg'` | `'md'` | Track height |

### Indicator

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `color` | `'default' \| 'secondary' \| 'destructive'` | `'default'` | Bar color |
